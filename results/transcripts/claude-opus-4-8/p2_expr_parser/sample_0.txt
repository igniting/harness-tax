"""Calc-26 expression evaluator."""

import math


def evaluate(expr: str) -> int:
    tokens = _tokenize(expr)
    parser = _Parser(tokens)
    result = parser.parse()
    return result


def _tokenize(expr):
    tokens = []
    i = 0
    n = len(expr)
    while i < n:
        c = expr[i]
        if c.isspace():
            i += 1
            continue
        if c.isdigit():
            j = i
            while j < n and expr[j].isdigit():
                j += 1
            tokens.append(('NUM', int(expr[i:j])))
            i = j
            continue
        if c in '+-*/^@()':
            tokens.append((c, c))
            i += 1
            continue
        raise ValueError(f"Unknown character: {c!r}")
    return tokens


class _Parser:
    def __init__(self, tokens):
        self.tokens = tokens
        self.pos = 0

    def peek(self):
        if self.pos < len(self.tokens):
            return self.tokens[self.pos]
        return (None, None)

    def advance(self):
        tok = self.tokens[self.pos]
        self.pos += 1
        return tok

    def parse(self):
        if not self.tokens:
            raise ValueError("Empty input")
        val = self.parse_at()
        if self.pos != len(self.tokens):
            raise ValueError("Unexpected trailing tokens")
        return val

    # Level 1: @ (left-assoc, lowest)
    def parse_at(self):
        val = self.parse_add()
        while self.peek()[0] == '@':
            self.advance()
            rhs = self.parse_add()
            val = _midpoint(val, rhs)
        return val

    # Level 2: + - (left-assoc)
    def parse_add(self):
        val = self.parse_mul()
        while self.peek()[0] in ('+', '-'):
            op = self.advance()[0]
            rhs = self.parse_mul()
            if op == '+':
                val = val + rhs
            else:
                val = val - rhs
        return val

    # Level 3: * / (left-assoc)
    def parse_mul(self):
        val = self.parse_unary()
        while self.peek()[0] in ('*', '/'):
            op = self.advance()[0]
            rhs = self.parse_unary()
            if op == '*':
                val = val * rhs
            else:
                if rhs == 0:
                    raise ZeroDivisionError("division by zero")
                val = val // rhs
        return val

    # Unary minus binds tighter than ^ 's left operand.
    # Handle unary here, then power.
    def parse_unary(self):
        if self.peek()[0] == '-':
            self.advance()
            operand = self.parse_unary()
            return self.parse_power_with_base(-operand)
        return self.parse_power()

    def parse_power(self):
        base = self.parse_atom()
        return self.parse_power_with_base(base)

    def parse_power_with_base(self, base):
        # Right-associative power. A unary minus directly after ^ is forbidden.
        if self.peek()[0] == '^':
            self.advance()
            if self.peek()[0] == '-':
                raise ValueError("Unary minus not allowed directly after '^'")
            exponent = self.parse_power()
            if exponent < 0:
                raise ValueError("Negative exponent")
            return base ** exponent
        return base

    def parse_atom(self):
        tok = self.peek()
        if tok[0] == 'NUM':
            self.advance()
            return tok[1]
        if tok[0] == '(':
            self.advance()
            val = self.parse_at()
            if self.peek()[0] != ')':
                raise ValueError("Unbalanced parentheses")
            self.advance()
            return val
        if tok[0] is None:
            raise ValueError("Unexpected end of input")
        raise ValueError(f"Unexpected token: {tok[0]!r}")


def _midpoint(a, b):
    return (a + b) // 2
