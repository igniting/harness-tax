from collections import deque


class RateLimiter:
    def __init__(self, global_limit: int, window_ms: int):
        self.global_limit = global_limit
        self.window_ms = window_ms
        # Per-key config: {key: (limit, weight)}
        self.key_config: dict[str, tuple[int, int]] = {}
        # Per-key history: {key: deque of (ts, weight)}
        self.key_history: dict[str, deque[tuple[int, int]]] = {}
        # Per-key running sum of weights in window
        self.key_usage: dict[str, int] = {}
        # Global history: deque of (ts, weight)
        self.global_history: deque[tuple[int, int]] = deque()
        # Global running sum
        self.global_usage: int = 0

    def _evict_key(self, key: str, ts: int) -> None:
        cutoff = ts - self.window_ms
        history = self.key_history[key]
        usage = self.key_usage[key]
        while history and history[0][0] <= cutoff:
            _, w = history.popleft()
            usage -= w
        self.key_usage[key] = usage

    def _evict_global(self, ts: int) -> None:
        cutoff = ts - self.window_ms
        while self.global_history and self.global_history[0][0] <= cutoff:
            _, w = self.global_history.popleft()
            self.global_usage -= w

    def set_key_limit(self, key: str, limit: int, weight: int = 1) -> None:
        self.key_config[key] = (limit, weight)
        if key not in self.key_history:
            self.key_history[key] = deque()
            self.key_usage[key] = 0

    def allow(self, key: str, ts: int) -> bool:
        # Rule a: unregistered key
        if key not in self.key_config:
            return False

        # Evict expired events
        self._evict_key(key, ts)
        self._evict_global(ts)

        limit, weight = self.key_config[key]

        # Rule d: check per-key limit
        if self.key_usage[key] + weight > limit:
            return False

        # Rule d: check global limit
        if self.global_usage + weight > self.global_limit:
            return False

        # Admitted: record event
        self.key_history[key].append((ts, weight))
        self.key_usage[key] += weight
        self.global_history.append((ts, weight))
        self.global_usage += weight

        return True

    def usage(self, ts: int) -> dict[str, int]:
        result: dict[str, int] = {}
        for key in self.key_config:
            self._evict_key(key, ts)
            result[key] = self.key_usage[key]
        # Also evict global while we're at it
        self._evict_global(ts)
        return result
